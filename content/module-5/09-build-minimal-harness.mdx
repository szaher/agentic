## Overview

You've learned the eight harness concerns conceptually across Lessons 1–8. Now you'll build a working harness from scratch.

**Design constraints:**
- No persistence (no database, no file checkpoints)
- No tracing infrastructure (no logs beyond print statements)
- No frameworks (no LangGraph, no LangChain, no AgentOps)

**Why this is valuable:** Understanding the bare-metal implementation makes framework abstractions legible. You'll see exactly what LangGraph's `StateGraph` or Claude SDK's `Runner` are doing under the hood.

## Stage 1: The Bare Loop

Start with the simplest possible loop: send a message, get a response, repeat.

```python
# harness_v1.py
class FakeModelClient:
    """Deterministic model stub for reproducible labs."""
    
    def __init__(self):
        self.turn_count = 0
    
    def create_message(self, messages, max_tokens=1024):
        self.turn_count += 1
        user_text = messages[-1]["content"] if messages else ""
        
        # Return a canned text response
        return {
            "id": f"msg_{self.turn_count}",
            "role": "assistant",
            "content": [
                {
                    "type": "text",
                    "text": f"I received: '{user_text}'. This is a fake response."
                }
            ],
            "usage": {"input_tokens": 50, "output_tokens": 20}
        }


def run_agent_v1(goal: str):
    """Version 1: Bare loop with no termination."""
    client = FakeModelClient()
    messages = [{"role": "user", "content": goal}]
    
    while True:
        response = client.create_message(messages, max_tokens=1024)
        content = response["content"][0]["text"]
        print(f"Turn {client.turn_count}: {content}")
        
        # No exit condition — loops forever!


# Run it
run_agent_v1("What is the capital of France?")
```

**What happens:** The fake client returns a canned response, the harness prints it, then loops again with the same message. It runs forever because there's no termination condition.

**What's missing:**
- No tools (the agent can't take actions)
- No termination signal (no way to stop)
- No context growth (same message repeated)

Let's fix that.

## Stage 2: Add Tool Dispatch

Now we add:
- A tool registry (Python dict mapping tool names to functions)
- Tool definitions with schemas
- Tool-call parsing from model responses
- Tool-result formatting back to the model
- A "done" tool that signals loop termination

```python
# harness_v2.py
import json
from typing import Dict, Callable, Any


class FakeModelClient:

# Tool Registry
class ToolRegistry:
    def __init__(self):
        self.tools: Dict[str, Callable] = {}
        self.schemas: Dict[str, dict] = {}
    
    def register(self, name: str, fn: Callable, schema: dict):
        """Register a tool with its implementation and schema."""
        self.tools[name] = fn
        self.schemas[name] = schema
    
    def dispatch(self, name: str, input_data: dict) -> Any:
        """Dispatch a tool call by name."""
        if name not in self.tools:
            raise ValueError(f"Tool '{name}' not registered")
        return self.tools[name](**input_data)  # Unpack dict as kwargs
    
    def get_tool_schemas(self):
        """Return all tool schemas for the API."""
        return list(self.schemas.values())


# Tool Implementations
def web_search(query: str) -> str:
    """Fake web search — returns canned results based on keywords."""
    if "population" in query.lower() and "france" in query.lower():
        return json.dumps({
            "results": [
                {"title": "France Population", "snippet": "France has ~67 million people."}
            ]
        })
    return json.dumps({"results": []})


def calculator(expression: str) -> str:
    """Simple calculator — evaluates basic math."""
    try:
        result = eval(expression, {"__builtins__": {}}, {})
        return str(result)
    except Exception as e:
        return f"Error: {e}"


# Global flag for termination
_done_flag = False

def done() -> str:
    """Signal that the agent is done."""
    global _done_flag
    _done_flag = True
    return "Task complete."


def run_agent_v2(goal: str):
    """Version 2: Add tool dispatch and termination."""
    global _done_flag
    _done_flag = False
    
    # Initialize client and registry
    client = FakeModelClient()
    registry = ToolRegistry()
    
    # Register tools
    registry.register("web_search", web_search, {
        "name": "web_search",
        "description": "Search the web for information",
        "input_schema": {
            "type": "object",
            "properties": {"query": {"type": "string"}},
            "required": ["query"]
        }
    })
    
    registry.register("calculator", calculator, {
        "name": "calculator",
        "description": "Evaluate a math expression",
        "input_schema": {
            "type": "object",
            "properties": {"expression": {"type": "string"}},
            "required": ["expression"]
        }
    })
    
    registry.register("done", done, {
        "name": "done",
        "description": "Signal task completion",
        "input_schema": {"type": "object", "properties": {}}
    })
    
    # Start the loop
    messages = [{"role": "user", "content": goal}]
    
    while not _done_flag:
        response = client.create_message(
            messages,
            tools=registry.get_tool_schemas(),
            max_tokens=1024
        )
        
        # Parse content blocks
        assistant_message = {"role": "assistant", "content": []}
        tool_results = []
        
        for block in response["content"]:
            if block["type"] == "text":
                print(f"[Turn {client.turn_count}] Assistant: {block['text']}")
                assistant_message["content"].append(block)
            
            elif block["type"] == "tool_use":
                tool_name = block["name"]
                tool_input = block["input"]
                tool_id = block["id"]
                
                print(f"[Turn {client.turn_count}] Calling tool: {tool_name}({tool_input})")
                
                # Dispatch the tool
                result = registry.dispatch(tool_name, tool_input)
                print(f"[Turn {client.turn_count}] Tool result: {result}")
                
                # Append tool_use block to assistant message
                assistant_message["content"].append(block)
                
                # Format tool_result
                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": tool_id,
                    "content": result
                })
        
        # Append assistant message to history
        messages.append(assistant_message)
        
        # Append tool results (if any) as a user message
        if tool_results:
            messages.append({"role": "user", "content": tool_results})
    
    print("\n[Agent stopped]")


# Run it
run_agent_v2("What is the population of France?")
```

**What happens:**
1. Turn 1: Model calls `web_search("population of France")`
2. Harness dispatches tool, gets canned result
3. Turn 2: Model returns text answer, calls `done()`
4. Loop terminates

**What's new:**
- Tool registry with schema validation
- Tool-call parsing from `content` blocks
- Tool-result formatting back to the API
- Termination signal via the `done()` tool

**What's still missing:**
- Context assembly is implicit (just appending to `messages`)
- No system prompt (the model doesn't know its role)
- No loop controls (max turns, token budgets)

Let's add context assembly.

## Stage 3: Add Context Assembly

Context assembly is the process of building the `messages` array for each turn. In a production harness, this includes:
- System prompt (role, goals, instructions)
- Conversation history (previous turns)
- Tool results (from the current turn)
- Working memory or session state

We'll extract this into a dedicated function.

```python
# harness_v3.py (changes only)

def assemble_context(goal: str, history: list) -> list:
    """Assemble the messages array for the next API call.
    
    Args:
        goal: The user's original goal
        history: List of prior assistant/user messages
    
    Returns:
        List of messages with system prompt + history
    """
    # System prompt
    system_message = {
        "role": "system",
        "content": (
            "You are a research assistant agent. "
            "Use the web_search tool to find information. "
            "Use the calculator tool for math. "
            "When you have a complete answer, call the 'done' tool to finish."
        )
    }
    
    # User goal
    user_message = {"role": "user", "content": goal}
    
    # Combine: system + user goal + history
    messages = [system_message, user_message] + history
    return messages


def run_agent_v3(goal: str):

    global _done_flag

    client = FakeModelClient()

                tool_input = block["input"]
                tool_id = block["id"]

                result = registry.dispatch(tool_name, tool_input)
                print(f"[Turn {client.turn_count}] Tool result: {result}")
```

**What changed:**
- Extracted `assemble_context()` function
- Added a system prompt explaining the agent's role and available tools
- Separated `history` from the initial `goal`
- Context is rebuilt from scratch each turn (system + goal + history)

**Why this matters:** In production, context assembly is where you inject:
- Updated memory or session state
- Dynamic instructions based on run state
- Prompt caching boundaries
- Token budget enforcement

This is the **most critical extension point** in a harness.

**Run it:** The behavior is identical to v2, but the code is now structured for growth.

## Stage 4: Add Loop Controls

Production harnesses need guardrails:
- **Max turns:** Prevent infinite loops
- **Token budgets:** Stop before exceeding cost limits
- **Timeout:** Fail fast if the agent is stuck

We'll add all three.

```python
# harness_v4.py (changes only)

def run_agent_v4(goal: str, max_turns: int = 10, max_tokens: int = 10000):
    """Version 4: Add loop controls (max turns, token budget, timeout)."""
    global _done_flag
    _done_flag = False
    
    client = FakeModelClient()
    registry = ToolRegistry()
    
    # [Tool registration identical to v2/v3 — omitted]
    
    history = []
    turn_count = 0
    cumulative_tokens = 0
    
    while not _done_flag:
        # Check loop controls
        if turn_count >= max_turns:
            print(f"\n[Warning] Reached max turns ({max_turns}). Stopping with partial result.")
            break
        
        if cumulative_tokens >= max_tokens:
            print(f"\n[Warning] Exceeded token budget ({max_tokens}). Stopping.")
            break
        
        # Assemble context
        messages = assemble_context(goal, history)
        
        response = client.create_message(
            messages,
            tools=registry.get_tool_schemas(),
            max_tokens=1024
        )
        
        # Track token usage
        usage = response.get("usage", {})
        turn_tokens = usage.get("input_tokens", 0) + usage.get("output_tokens", 0)
        cumulative_tokens += turn_tokens
        
        turn_count += 1
        print(f"\n[Turn {turn_count}] Tokens this turn: {turn_tokens} | Cumulative: {cumulative_tokens}")
        
        # Parse content blocks
        assistant_message = {"role": "assistant", "content": []}
        tool_results = []
        
        for block in response["content"]:
            if block["type"] == "text":
                print(f"  Assistant: {block['text']}")
                assistant_message["content"].append(block)
            
            elif block["type"] == "tool_use":
                tool_name = block["name"]
                tool_input = block["input"]
                tool_id = block["id"]
                
                print(f"  Calling tool: {tool_name}({tool_input})")
                result = registry.dispatch(tool_name, tool_input)
                print(f"  Tool result: {result}")
                
                assistant_message["content"].append(block)
                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": tool_id,
                    "content": result
                })
        
        # Append to history
        history.append(assistant_message)
        if tool_results:
            history.append({"role": "user", "content": tool_results})
    
    print(f"\n[Agent stopped] Total turns: {turn_count}, Total tokens: {cumulative_tokens}")


# Run with max_turns constraint
run_agent_v4("What is the population of France?", max_turns=3)
```

**What changed:**
- Added `max_turns` and `max_tokens` parameters
- Track `turn_count` and `cumulative_tokens`
- Check limits at the start of each loop iteration
- Print token usage per turn
- Break with a warning if limits are exceeded

**Run it with constraints:** Try `run_agent_v4("What is the capital of Germany?", max_turns=1)`. The agent will call one tool, then stop mid-task with a partial result.

**Why this matters:** In production, agents can get stuck in loops (tool → error → retry → error). Loop controls prevent runaway costs and infinite execution.

**What's still missing:**
- State sharing between tools (tools are stateless functions)
- Checkpointing (no way to pause/resume)
- Tracing (no structured logs)
- Approval gates (no human intervention)

You'll add these in Lesson 10.

```python
# harness.py
import json
from typing import Dict, Callable, Any, List


class FakeModelClient:
    """Deterministic model stub for reproducible labs.
    
    In production, replace this with:
        from anthropic import Anthropic
        client = Anthropic(api_key="...")
    
    Or:
        from openai import OpenAI
        client = OpenAI(api_key="...")
    """
    
    def __init__(self):
        self.turn_count = 0
    
    def create_message(self, messages, tools=None, max_tokens=1024):
        """Scripted responses based on keywords in the last user message."""
        self.turn_count += 1
        
        # Extract last user message
        user_text = ""
        for msg in reversed(messages):
            if msg["role"] == "user":
                if isinstance(msg["content"], str):
                    user_text = msg["content"]
                break
        
        # Script 1: population question → web_search
        if "population" in user_text.lower() and self.turn_count == 1:
            return {
                "id": f"msg_{self.turn_count}",
                "role": "assistant",
                "content": [
                    {
                        "type": "tool_use",
                        "id": f"tool_call_{self.turn_count}",
                        "name": "web_search",
                        "input": {"query": "population of France"}
                    }
                ],
                "usage": {"input_tokens": 100, "output_tokens": 30}
            }
        
        # Script 2: after web_search result → answer + done
        if self.turn_count == 2:
            return {
                "id": f"msg_{self.turn_count}",
                "role": "assistant",
                "content": [
                    {
                        "type": "text",
                        "text": "Based on the search, France has approximately 67 million people."
                    },
                    {
                        "type": "tool_use",
                        "id": f"tool_call_{self.turn_count}",
                        "name": "done",
                        "input": {}
                    }
                ],
                "usage": {"input_tokens": 150, "output_tokens": 40}
            }
        
        # Script 3: math question → calculator
        if "calculate" in user_text.lower() or "math" in user_text.lower():
            if self.turn_count == 1:
                return {
                    "id": f"msg_{self.turn_count}",
                    "role": "assistant",
                    "content": [
                        {
                            "type": "tool_use",
                            "id": f"tool_call_{self.turn_count}",
                            "name": "calculator",
                            "input": {"expression": "123 * 456"}
                        }
                    ],
                    "usage": {"input_tokens": 80, "output_tokens": 25}
                }
            else:
                return {
                    "id": f"msg_{self.turn_count}",
                    "role": "assistant",
                    "content": [
                        {
                            "type": "text",
                            "text": "The result is shown above."
                        },
                        {
                            "type": "tool_use",
                            "id": f"tool_call_{self.turn_count}",
                            "name": "done",
                            "input": {}
                        }
                    ],
                    "usage": {"input_tokens": 100, "output_tokens": 20}
                }
        
        # Default: return "I don't know" and call done
        return {
            "id": f"msg_{self.turn_count}",
            "role": "assistant",
            "content": [
                {"type": "text", "text": "I don't have enough information."},
                {
                    "type": "tool_use",
                    "id": f"tool_call_{self.turn_count}",
                    "name": "done",
                    "input": {}
                }
            ],
            "usage": {"input_tokens": 50, "output_tokens": 15}

class ToolRegistry:
    """Tool registry with schema validation and dispatch."""
    
    def __init__(self):
        self.tools: Dict[str, Callable] = {}
        self.schemas: Dict[str, dict] = {}
    
    def register(self, name: str, fn: Callable, schema: dict):
        """Register a tool with its implementation and schema."""
        self.tools[name] = fn
        self.schemas[name] = schema
    
    def dispatch(self, name: str, input_data: dict) -> Any:
        """Dispatch a tool call by name."""
        if name not in self.tools:
            raise ValueError(f"Tool '{name}' not registered")
        return self.tools[name](**input_data)  # Unpack dict as kwargs
    
    def get_tool_schemas(self) -> List[dict]:
        """Return all tool schemas for the API."""
        return list(self.schemas.values())


class Harness:
    """Minimal agent harness with loop, tools, context, and controls."""
    
    def __init__(self, client, registry: ToolRegistry, max_turns: int = 10, max_tokens: int = 10000):
        self.client = client
        self.registry = registry
        self.max_turns = max_turns
        self.max_tokens = max_tokens
        self.done_flag = False
    
    def assemble_context(self, goal: str, history: List[dict]) -> List[dict]:
        """Assemble messages for the next API call."""
        system_message = {
            "role": "system",
            "content": (
                "You are a research assistant agent. "
                "Use the web_search tool to find information. "
                "Use the calculator tool for math. "
                "When you have a complete answer, call the 'done' tool to finish."
            )
        }
        
        user_message = {"role": "user", "content": goal}
        
        # System + goal + history
        return [system_message, user_message] + history
    
    def run(self, goal: str) -> dict:
        """Run the agent loop until done or limits reached."""
        history = []
        turn_count = 0
        cumulative_tokens = 0
        
        print(f"[Harness] Starting agent with goal: '{goal}'\n")
        
        while not self.done_flag:
            # Check loop controls
            if turn_count >= self.max_turns:
                print(f"\n[Warning] Reached max turns ({self.max_turns}). Stopping.")
                break
            
            if cumulative_tokens >= self.max_tokens:
                print(f"\n[Warning] Exceeded token budget ({self.max_tokens}). Stopping.")
                break
            
            # Assemble context
            messages = self.assemble_context(goal, history)
            
            # Call the model
            response = self.client.create_message(
                messages,
                tools=self.registry.get_tool_schemas(),
                max_tokens=1024
            )
            
            # Track token usage
            usage = response.get("usage", {})
            turn_tokens = usage.get("input_tokens", 0) + usage.get("output_tokens", 0)
            cumulative_tokens += turn_tokens
            turn_count += 1
            
            print(f"[Turn {turn_count}] Tokens: {turn_tokens} | Cumulative: {cumulative_tokens}")
            
            # Parse content blocks
            assistant_message = {"role": "assistant", "content": []}
            tool_results = []
            
            for block in response["content"]:
                if block["type"] == "text":
                    print(f"  Assistant: {block['text']}")
                    assistant_message["content"].append(block)
                
                elif block["type"] == "tool_use":
                    tool_name = block["name"]
                    tool_input = block["input"]
                    tool_id = block["id"]
                    
                    print(f"  Calling tool: {tool_name}({tool_input})")
                    
                    # Dispatch the tool
                    result = self.registry.dispatch(tool_name, tool_input)
                    print(f"  Tool result: {result}")
                    
                    assistant_message["content"].append(block)
                    tool_results.append({
                        "type": "tool_result",
                        "tool_use_id": tool_id,
                        "content": result
                    })
            
            # Append to history
            history.append(assistant_message)
            if tool_results:
                history.append({"role": "user", "content": tool_results})
        
        print(f"\n[Harness] Agent stopped. Total turns: {turn_count}, Total tokens: {cumulative_tokens}")
        
        return {
            "turns": turn_count,
            "tokens": cumulative_tokens,
            "history": history
        }

        return str(result)
    except Exception as e:
        return f"Error: {e}"


def done() -> str:
    """Signal that the agent is done."""
    # In production, this would set a flag on the Harness instance
    # For now, we use a global flag (see main())
    return "Task complete."


# Main
def main():
    # Initialize client and registry
    client = FakeModelClient()
    registry = ToolRegistry()
    
    # Register tools
    registry.register("web_search", web_search, {
        "name": "web_search",
        "description": "Search the web for information",
        "input_schema": {
            "type": "object",
            "properties": {"query": {"type": "string", "description": "Search query"}},
            "required": ["query"]
        }
    })
    
    registry.register("calculator", calculator, {
        "name": "calculator",
        "description": "Evaluate a mathematical expression",
        "input_schema": {
            "type": "object",
            "properties": {"expression": {"type": "string", "description": "Math expression to evaluate"}},
            "required": ["expression"]
        }
    })
    
    # Create a done callback that sets the harness flag
    harness = Harness(client, registry, max_turns=10, max_tokens=10000)
    
    def done_callback() -> str:
        harness.done_flag = True
        return "Task complete."
    
    registry.register("done", done_callback, {
        "name": "done",
        "description": "Signal task completion",
        "input_schema": {"type": "object", "properties": {}}
    })
    
    # Run the agent
    result = harness.run("What is the population of France?")
    
    print("\n" + "="*60)
    print("FINAL RESULT:")
    print(f"  Turns: {result['turns']}")
    print(f"  Tokens: {result['tokens']}")
    print("="*60)


if __name__ == "__main__":
    main()
```

**What you have now:**
- A ~150-line `Harness` class with `run()` and `assemble_context()`
- A `ToolRegistry` with schema validation and dispatch
- Three tools: `web_search`, `calculator`, `done`
- Loop controls: max turns, token budget
- Complete, runnable code (no external dependencies except Python 3.8+)
